SELECT IL.NAME, I.DESCRIPTION
FROM ISSUES I
JOIN ISSUESLISTS IL ON IL.ID = I.ISSUESLIST_ID;

SELECT  I.DESCRIPTION, U.LASTNAME, U.FIRSTNAME
FROM ISSUES I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_OF_TASKS
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.LASTNAME
HAVING COUNT(*)>0;
